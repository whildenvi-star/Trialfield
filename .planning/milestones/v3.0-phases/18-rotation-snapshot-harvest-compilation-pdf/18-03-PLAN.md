    ```typescript
    const hasCurrentYearEnterprises = fields.some(f =>

    );

    // Check for SYNCED material usages by iterating the already-loaded enterprises
    const hasSyncedInputs = farm.fields.some(f =>

        e.materialUsages.some(mu => mu.dataSource === 'SYNCED')
      )
    );

    // Check for SYNCED harvest events
    const hasSyncedHarvest = farm.fields.some(f =>
      f.enterprises.some(e =>
        e.cropYear === cropYear &&
        e.harvestEvents.some(he => he.dataSource === 'SYNCED')
      )
    );

    // Snapshot requires one additional lightweight count query (not in the main farm query)
    const snapshotCount = await prisma.fieldHistory.count({
      where: { year: cropYear, field: { farmId } },
    });

    // Seed check: look for SeedLot records associated with current year enterprises
    // SeedUsage is not directly on the enterprise include chain, so check via a count
    const seedCount = await prisma.seedUsage.count({
      where: {
        fieldEnterprise: {
          field: { farmId },
          cropYear,
        },
        dataSource: 'SYNCED',
      },
    });

    const compileChecklist: CompileChecklist = {
      fields: fields.length > 0,
      enterprises: hasCurrentYearEnterprises,
      inputs: hasSyncedInputs,
      seeds: seedCount > 0,
      harvest: hasSyncedHarvest,
      snapshot: snapshotCount > 0,
    };
    ```

    **2. harvest-log.tsx — Add null guard for harvest.date:**

    Currently line 98 has `{format(harvest.date, "MM/dd/yyyy")}`. The `HarvestRecord.date` type is `Date` (non-nullable in TypeScript), but at runtime a SYNCED HarvestEvent could have an edge-case null date if the grain-tickets delivery date was missing.

    Add a null guard in the date column rendering:

    Change:
    ```tsx
    <Text style={col("12%")}>
      {format(harvest.date, "MM/dd/yyyy")}
    </Text>
    ```

    To:
    ```tsx
    <Text style={col("12%")}>
      {harvest.date ? format(harvest.date, "MM/dd/yyyy") : "—"}
    </Text>
    ```

    Also add a null guard for `harvest.acresHarvested` (could be 0, which is fine, but guard against undefined):
    ```tsx
    <Text style={col("8%")}>

    ```

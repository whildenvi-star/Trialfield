**Create the maps page — `src/app/(protected)/app/maps/page.tsx`:**

This is a Server Component that SSR-prefetches the farm center config, then passes it to the client map component.

```typescript
import { Suspense } from 'react'
import { createClient } from '@/lib/supabase/server'
import { FieldMap } from '@/components/maps/field-map'
import { DEFAULT_MAP_CENTER, DEFAULT_MAP_ZOOM } from '@/lib/map-config'

export default async function MapsPage() {
  const supabase = await createClient()

  // Prefetch farm center from config table
  let mapCenter: [number, number] = DEFAULT_MAP_CENTER
  let mapZoom = DEFAULT_MAP_ZOOM

  try {
    const { data } = await supabase
      .from('farm_map_config')
      .select('value')
      .eq('key', 'farm_center')
      .single()
    if (data?.value) {
      const cfg = data.value as { lat: number; lng: number; zoom: number }
      mapCenter = [cfg.lng, cfg.lat]
      mapZoom = cfg.zoom ?? DEFAULT_MAP_ZOOM
    }
  } catch {
    // Use defaults if no config exists
  }

  return (
    <div className="fixed inset-0 top-[48px]">  {/* top-[48px] clears portal top bar */}
      <Suspense fallback={<div className="w-full h-full bg-[#080604] animate-pulse" />}>
        <FieldMap initialCenter={mapCenter} initialZoom={mapZoom} />
      </Suspense>
    </div>
  )
}
```

Note: The map must be `dynamic` (no SSR) because Mapbox GL JS accesses `window`. The page is a server component that renders a client component. The `FieldMap` client component will use `'use client'` and `dynamic` import internally.

**Open-Meteo GDD fetch:**
```typescript
useEffect(() => {
  if (!field?.centroid_lat || !field?.centroid_lng) return
  setGddLoading(true)
  setGddData(null)
  
  const today = new Date()
  const end = today.toISOString().split('T')[0]
  const start = new Date(today.getTime() - 6 * 86400000).toISOString().split('T')[0]
  
  const url = `https://api.open-meteo.com/v1/forecast?latitude=${field.centroid_lat}&longitude=${field.centroid_lng}&daily=et0_fao_evapotranspiration&temperature_unit=fahrenheit&start_date=${start}&end_date=${end}&timezone=America%2FChicago`
  
  // Note: Open-Meteo doesn't have a direct growing_degree_days variable in forecast API.
  // Use temperature_2m_max and temperature_2m_min to compute GDD base 50°F:
  // GDD_day = max(0, ((T_max + T_min) / 2) - 50)
  
  const gddUrl = `https://api.open-meteo.com/v1/forecast?latitude=${field.centroid_lat}&longitude=${field.centroid_lng}&daily=temperature_2m_max,temperature_2m_min&temperature_unit=fahrenheit&start_date=${start}&end_date=${end}&timezone=America%2FChicago`
  
  fetch(gddUrl)
    .then(r => r.json())
    .then(data => {
      const dates = data.daily?.time ?? []
      const maxTemps = data.daily?.temperature_2m_max ?? []
      const minTemps = data.daily?.temperature_2m_min ?? []
      const days = dates.map((date: string, i: number) => {
        const avg = (maxTemps[i] + minTemps[i]) / 2
        const gdd = Math.max(0, avg - 50)
        return { date, gdd: Math.round(gdd * 10) / 10 }
      })
      setGddData(days)
    })
    .catch(() => setGddError(true))
    .finally(() => setGddLoading(false))
}, [field?.registry_field_id])
```

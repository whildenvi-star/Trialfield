**Upload handler:**
```typescript
async function handleDrop(files: File[]) {
  const file = files[0]
  if (!file) return
  if (!file.name.endsWith('.zip')) {
    setError('Only .zip shapefile bundles are accepted')
    setStatus('error')
    return
  }
  setStatus('uploading')
  setResult(null)

  const formData = new FormData()
  formData.append('file', file)

  try {
    const res = await fetch('/api/maps/import', { method: 'POST', body: formData })
    const data = await res.json()
    if (!res.ok) {
      setError(data.error ?? 'Import failed')
      setStatus('error')
      return
    }
    setResult(data)
    setStatus('complete')
  } catch {
    setError('Network error — import failed')
    setStatus('error')
  }
}
```

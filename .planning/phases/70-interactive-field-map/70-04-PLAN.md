**Create `src/app/(protected)/admin/maps-import/page.tsx`:**

Server Component. Auth gate: use `createClient()`, get session user, fetch profile role. If role !== 'admin', `redirect('/app')`. If no session, `redirect('/auth/login')`.

```typescript
import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { BoundaryImport } from '@/components/maps/boundary-import'

export default async function MapsImportPage() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/auth/login')
  
  const { data: profile } = await supabase
    .from('profiles')
    .select('role')
    .eq('id', user.id)
    .single()
  
  if (profile?.role !== 'admin') redirect('/app')
  
  return (
    <div className="max-w-2xl mx-auto px-6 py-10">
      <h1 className="font-mono text-xl text-[#C8860A] mb-2">Field Boundary Import</h1>
      <p className="text-[#6a5a4a] text-sm mb-8">
        Upload a GeoJSON or KML export from SMS to load or update field boundary polygons.
        Fields are matched by name. Re-importing a field overwrites its existing boundary.
      </p>
      <BoundaryImport />
    </div>
  )
}
```

**Upload handler:**
```typescript
async function handleDrop(files: File[]) {
  const file = files[0]
  if (!file) return
  setStatus('uploading')
  setResult(null)
  
  const formData = new FormData()
  formData.append('file', file)
  
  try {
    const res = await fetch('/api/maps/import', { method: 'POST', body: formData })
    const data = await res.json()
    if (!res.ok) {
      setError(data.error ?? 'Import failed')
      setStatus('error')
      return
    }
    setResult(data)
    setStatus('complete')
  } catch (err) {
    setError('Network error — import failed')
    setStatus('error')
  }
}
```

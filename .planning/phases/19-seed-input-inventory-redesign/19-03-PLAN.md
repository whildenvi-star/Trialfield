```javascript
  '@media print { body { -webkit-print-color-adjust: exact; } }'
].join('\\n');
```

`openPrintWindow(title, bodyHtml)` helper:
```javascript
function openPrintWindow(title, bodyHtml) {
  var win = window.open('', '_blank');
  if (!win) { util.showToast('Enable popups to print reports', 4000, 'error'); return; }
  win.document.write('<!DOCTYPE html><html><head><title>' + util.escHtml(title) + '</title><style>' + PRINT_CSS + '</style></head><body>' + bodyHtml + '</body></html>');
  win.document.close();
  win.focus();
  win.print();
}
```
CRITICAL: Call `window.open()` synchronously in the event handler (not after a Promise) to avoid popup blockers.

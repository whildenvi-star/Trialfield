     include: {

           fieldEnterprise: {

             cropYear: year,
           },
         },
         include: {

               field: { select: { name: true } },

4. Filter to only seed lots that have usages in the target year, then map to SeedLotRecord:
   ```typescript
   import { checkSeedCompliance } from "@/lib/compile/nop-compliance";

   const seedLotRecords: SeedLotRecord[] = seedLots
     .filter(sl => sl.seedUsages.length > 0)
     .map(sl => {
       const compliance = checkSeedCompliance({
         id: sl.id,
         variety: sl.variety,
         isOrganic: sl.isOrganic,
         isUntreated: sl.isUntreated,
         commercialAvailDoc: sl.commercialAvailDoc,
       });
       const fieldNames = [...new Set(sl.seedUsages.map(u => u.fieldEnterprise.field.name))];
       const totalAcres = sl.seedUsages.reduce((sum, u) => sum + u.acres, 0);
       return {
         id: sl.id,
         crop: sl.crop,
         variety: sl.variety,
         brand: sl.brand,
         supplier: sl.supplier,
         certNumber: sl.certNumber,
         lotNumber: sl.lotNumber,
         isOrganic: sl.isOrganic,
         isUntreated: sl.isUntreated,
         omriListed: false,  // Will be true when seed-inventory data populates this via compile
         commercialAvailDoc: sl.commercialAvailDoc,
         fieldNames,
         totalAcres,
         verdict: compliance.verdict,
         reasons: compliance.reasons,
       };
     });
   ```

5. Include seedLotRecords in the returned report data object: `seedLots: seedLotRecords`

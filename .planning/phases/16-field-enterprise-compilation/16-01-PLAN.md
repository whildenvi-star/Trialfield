```typescript
import type { BudgetEnterprise, BudgetField } from "@/lib/ecosystem/types";

export interface OrganicEnterprise {
  enterprise: BudgetEnterprise;
  fields: BudgetField[];
}

export function filterOrganicEnterprises(
  enterprises: BudgetEnterprise[],
  fields: BudgetField[]
): OrganicEnterprise[] {
  const organic = enterprises.filter((e) => e.category === "organic");
  const organicIds = new Set(organic.map((e) => e.id));
  return organic.map((e) => ({
    enterprise: e,
    fields: fields.filter((f) => organicIds.has(f.enterpriseId)),
  }));
}
```

**3. Create src/lib/compile/field-mapper.ts:**

```typescript
export interface FieldMatch {
  fieldId: string;
  fieldName: string;
  matchMethod: "name" | "alias" | "stored-mapping";
}

export function resolveField(
  budgetFieldName: string,
  localFields: Array<{ id: string; name: string; farmBudgetFieldName: string | null }>,
  registryAliasMap: Map<string, string>  // lowercase alias -> field id
): FieldMatch | null {
  const lc = budgetFieldName.toLowerCase();

  // 1. Exact name match (case-insensitive)
  const byName = localFields.find((f) => f.name.toLowerCase() === lc);
  if (byName) return { fieldId: byName.id, fieldName: byName.name, matchMethod: "name" };

  // 2. Registry alias match
  const aliasId = registryAliasMap.get(lc);
  if (aliasId) {
    const byAlias = localFields.find((f) => f.id === aliasId);
    if (byAlias) return { fieldId: byAlias.id, fieldName: byAlias.name, matchMethod: "alias" };
  }

  // 3. Stored mapping (farmBudgetFieldName persisted from prior manual resolution)
  const byStored = localFields.find(
    (f) => f.farmBudgetFieldName?.toLowerCase() === lc
  );
  if (byStored) return { fieldId: byStored.id, fieldName: byStored.name, matchMethod: "stored-mapping" };

  return null;
}
```

**4. Create src/lib/compile/compile-engine.ts:**

This is the core orchestrator. It:

a) Fetches from all three ecosystem sources using `Promise.allSettled`:
   - `getBudgetOrganicFields()` from budget-client (returns { enterprises, fields })
   - `getRegistryFields()` from registry-client
   - `getTicketsForCropYear(cropYear)` from tickets-client

b) Fetches local data from Prisma:
   - All farm fields with `farmBudgetFieldName` column: `prisma.field.findMany({ where: { farmId }, select: { id, name, farmBudgetFieldName, registryId } })`
   - All existing FieldEnterprise rows for this cropYear: `prisma.fieldEnterprise.findMany({ where: { field: { farmId }, cropYear }, select: { id, fieldId, crop, plantedAcres, label } })`
   - The farm record (first farm): `prisma.farm.findFirst()`

c) Builds a registry alias map: For each RegistryField, map each alias (lowercased) to the local Field that has `registryId === registryField.id`. This bridges registry aliases to organic-cert Field.id.

d) Runs nop-filter on the budget data.

e) For each organic budget field (from OrganicEnterprise[].fields), calls `resolveField()`:
   - If matched (fieldId not null): Look up existing FieldEnterprise for this fieldId+cropYear+crop+null-label.
     - If existing enterprise found, compare crop and plantedAcres. If identical → status="unchanged". If different → status="update" with diff.
     - If no existing enterprise → status="new".
   - If not matched → status="unmatched", fieldId=null.

   Use registry acres where available: for matched fields, look up the registry field by registryId and use `reportingAcres` as `registryAcres`. The `plantedAcres` value comes from the budget field.

f) Builds deliveries: Group tickets by `ticket.farm.toLowerCase()`, match to resolved fields by name (case-insensitive). For each matched field, produce a `TicketSummary` with totalLoads, totalLbs, and ticket brief list. Unmatched deliveries (ticket.farm doesn't match any organic-cert field) are silently excluded from the per-field view (they aren't relevant to NOP).

g) Builds readiness: For each organic-status Field (organic or transitional), check if any FieldEnterprise exists for this cropYear → enterprises="compiled" or "missing". Inputs and seeds are always "pending" (Phase 17).

h) Collects savedMappings: All fields where `farmBudgetFieldName IS NOT NULL`.

i) Returns a `CompilePreview` object.

IMPORTANT: `buildPreview()` performs ZERO database writes. It is purely read-only.

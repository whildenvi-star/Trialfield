```python
def log_detection(self, frame_id: int, detections: Dict[str, Any],
                  frame=None, log_frames_dir: Path = None) -> None:
    image_path = None
    if frame is not None and log_frames_dir is not None:
        from datetime import datetime
        date_dir = log_frames_dir / datetime.utcnow().strftime("%Y-%m-%d")
        date_dir.mkdir(parents=True, exist_ok=True)
        image_path = date_dir / f"frame_{frame_id:08d}.jpg"
        cv2.imwrite(str(image_path), frame, [cv2.IMWRITE_JPEG_QUALITY, 85])

    self.detection_queue.put({
        'timestamp': time(),
        'frame_id': frame_id,
        'detections': detections,
        'image_path': str(image_path) if image_path else None
    })
```

This is backward-compatible — existing callers without frame/log_frames_dir still work (image_path will be None).

3. Add `purge_old_frames()` method:
```python
def purge_old_frames(self, log_frames_dir: Path, retention_days: int) -> None:
    """Delete frame log directories older than retention_days."""
    if not log_frames_dir.exists():
        return
    from datetime import datetime, timedelta
    cutoff = datetime.utcnow() - timedelta(days=retention_days)
    for date_dir in log_frames_dir.iterdir():
        if date_dir.is_dir():
            try:
                dir_date = datetime.strptime(date_dir.name, "%Y-%m-%d")
                if dir_date < cutoff:
                    import shutil
                    shutil.rmtree(date_dir)
                    self.logger.info(f"Purged old frame log: {date_dir}")
            except ValueError:
                pass  # Not a date directory, skip
```
